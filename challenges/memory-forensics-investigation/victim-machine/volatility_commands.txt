ESSENTIAL VOLATILITY2 COMMANDS FOR THIS CASE

# Basic system information
volatility -f memory.raw imageinfo
volatility -f memory.raw --profile=Win10x64_19041 pslist
volatility -f memory.raw --profile=Win10x64_19041 pstree

# Network analysis
volatility -f memory.raw --profile=Win10x64_19041 netscan
volatility -f memory.raw --profile=Win10x64_19041 connscan

# Process analysis
volatility -f memory.raw --profile=Win10x64_19041 cmdline
volatility -f memory.raw --profile=Win10x64_19041 dlllist -p 3724
volatility -f memory.raw --profile=Win10x64_19041 handles -p 3724

# Memory extraction
volatility -f memory.raw --profile=Win10x64_19041 memdump -p 3892 -D output/
volatility -f memory.raw --profile=Win10x64_19041 procdump -p 3724 -D output/

# Registry analysis
volatility -f memory.raw --profile=Win10x64_19041 hivelist
volatility -f memory.raw --profile=Win10x64_19041 printkey -K "Software\Microsoft\Windows\CurrentVersion\Run"

# String analysis
strings output/3892.dmp | grep -E "[A-Za-z0-9+/]{20,}==?" | head -10
