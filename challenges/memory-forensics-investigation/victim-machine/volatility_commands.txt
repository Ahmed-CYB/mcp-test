COMPLETE VOLATILITY ANALYSIS COMMANDS
=====================================

# Step 1: Determine OS profile
vol.py -f memory.raw imageinfo

# Step 2: List all processes
vol.py -f memory.raw --profile=Win7SP1x64 pslist

# Step 3: Find hidden/terminated processes
vol.py -f memory.raw --profile=Win7SP1x64 psscan

# Step 4: Extract command line arguments
vol.py -f memory.raw --profile=Win7SP1x64 cmdline

# Step 5: Analyze network connections
vol.py -f memory.raw --profile=Win7SP1x64 netscan

# Step 6: Dump suspicious process memory (PID 2468)
vol.py -f memory.raw --profile=Win7SP1x64 memdump -p 2468 -D ./dumps/

# Step 7: Extract strings from process dump
strings dumps/2468.dmp | grep -E "CTF\{|flag|password"

# Step 8: Check for code injection
vol.py -f memory.raw --profile=Win7SP1x64 malfind

# Step 9: Examine registry entries
vol.py -f memory.raw --profile=Win7SP1x64 printkey -K "Software\Microsoft\Windows\CurrentVersion\Run"

# Step 10: Search for PowerShell artifacts
vol.py -f memory.raw --profile=Win7SP1x64 consoles
