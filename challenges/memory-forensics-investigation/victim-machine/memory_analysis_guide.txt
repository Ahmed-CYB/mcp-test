VOLATILITY2 MEMORY ANALYSIS WORKFLOW

STEP 1: Identify OS Profile
volatility -f memory.raw imageinfo

STEP 2: List Running Processes
volatility -f memory.raw --profile=Win10x64_19041 pslist

STEP 3: Analyze Process Tree
volatility -f memory.raw --profile=Win10x64_19041 pstree

STEP 4: Check Network Connections
volatility -f memory.raw --profile=Win10x64_19041 netscan

STEP 5: Extract Command Lines
volatility -f memory.raw --profile=Win10x64_19041 cmdline

STEP 6: Dump Suspicious Process Memory
volatility -f memory.raw --profile=Win10x64_19041 memdump -p [PID] -D output/

STEP 7: Search for Strings in Process Memory
strings output/[PID].dmp | grep -i "CTF{"

STEP 8: Analyze Registry Modifications
volatility -f memory.raw --profile=Win10x64_19041 printkey -K "Software\Microsoft\Windows\CurrentVersion\Run"
